#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, e, i;
    int *p;

    printf("Enter initial number of employees: ");
    scanf("%d", &n);

    p = (int *)calloc(n, sizeof(int));

    if (p == NULL) {
        printf("Memory allocation failed\n");
        return 0;
    }

    printf("\nInitial employee IDs (default values):\n");
    for (i = 0; i < n; i++) {
        printf("Employee %d ID: %d\n", i + 1, p[i]);
    }

    printf("\nEnter employee IDs:\n");
    for (i = 0; i < n; i++) {
        printf("Employee %d ID: ", i + 1);
        scanf("%d", &p[i]);
    }

    printf("\nEnter new total number of employees: ");
    scanf("%d", &e);

    int *temp = realloc(p, e * sizeof(int));
    if (temp == NULL) {
        printf("Memory reallocation failed\n");
        free(p);
        return 0;
    }
    p = temp;

    if (e > n) {
        for (i = n; i < e; i++) {
            p[i] = 0;
        }
    }

    printf("\nEnter IDs for new employees:\n");
    for (i = n; i < e; i++) {
        printf("Employee %d ID: ", i + 1);
        scanf("%d", &p[i]);
    }

    printf("\nUpdated Employee ID List:\n");
    for (i = 0; i < e; i++) {
        printf("Employee %d ID: %d\n", i + 1, p[i]);
    }

   
    free(p);
    printf("\nMemory free successfully\n");

    return 0;
}
