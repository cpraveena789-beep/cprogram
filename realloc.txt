#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, newstrength, i;
    int *array;
    int sum = 0;

    printf("Enter the number of elements: ");
    scanf("%d", &n);

    array = (int *)calloc(n, sizeof(int));

    if (array == NULL) {
        printf("Memory not allocated\n");
        return 0;
    }

    printf("Memory allocated for %d elements\n", n);

    printf("Enter %d numbers:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &array[i]);
        sum += array[i];
    }

    printf("Sum of elements before resizing = %d\n", sum);

    printf("Enter the new total size: ");
    scanf("%d", &newstrength);

    int *temp = realloc(array, newstrength * sizeof(int));
    if (temp == NULL) {
        printf("Memory reallocation failed\n");
        free(array);
        return 0;
    }
    array = temp;

    if (newstrength > n) {
        printf("Enter %d more elements:\n", newstrength - n);
        for (i = n; i < newstrength; i++) {
            scanf("%d", &array[i]);
            sum += array[i];
        }
    }

    printf("Final sum of elements = %d\n", sum);

    free(array);
    printf("Memory deleted successfully\n");

    return 0;
}
